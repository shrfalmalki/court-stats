    function setQuickFilter(page, period) {

      const today = new Date();

      const todayStr = today.toISOString().split('T')[0];

      

      let fromDate = '';

      let toDate = todayStr;

      

      if (period === 'today') {

        fromDate = todayStr;

        toDate = todayStr;

      } else if (period === 'yesterday') {

        const yesterday = new Date(today);

        yesterday.setDate(yesterday.getDate() - 1);

        const yesterdayStr = yesterday.toISOString().split('T')[0];

        fromDate = yesterdayStr;

        toDate = yesterdayStr;

      } else if (period === 'week') {

        const weekAgo = new Date(today);

        weekAgo.setDate(weekAgo.getDate() - 7);

        fromDate = weekAgo.toISOString().split('T')[0];

        toDate = todayStr;

      } else if (period === 'all') {

        fromDate = '';

        toDate = '';

      }

      

      if (page === 'chart') {

        document.getElementById('chart-date-from').value = fromDate;

        document.getElementById('chart-date-to').value = toDate;

        updateCharts();

      } else if (page === 'comprehensive') {

        document.getElementById('comprehensive-date-from').value = fromDate;

        document.getElementById('comprehensive-date-to').value = toDate;

        updateComprehensiveStats();

      }

    }

      });

        descriptionFrequency[record.description] = (descriptionFrequency[record.description] || 0) + 1;

      });

      });

      

      let mostCommonCapacity = '-';

    async function deleteRecord(backendId, buttonElement) {

      const record = allData.find(r => r.__backendId === backendId);

      if (!record) return;


      const originalText = buttonElement.innerHTML;

      

      buttonElement.innerHTML = '✓ تأكيد الحذف؟';

      buttonElement.style.background = '#dc3545';

      

      const handleConfirm = async () => {

        buttonElement.disabled = true;

        buttonElement.innerHTML = '<span class="loading"></span>';

        

        if (window.dataSdk) {

          const result = await window.dataSdk.delete(record);

          if (!result.isOk) {

            buttonElement.innerHTML = '❌ فشل الحذف';

            setTimeout(() => {

              buttonElement.innerHTML = originalText;

              buttonElement.disabled = false;

            }, 2000);

          }

        }

        

        document.removeEventListener('click', handleCancel);

      };

      

      const handleCancel = (e) => {

        if (e.target !== buttonElement && !buttonElement.contains(e.target)) {

          buttonElement.innerHTML = originalText;

          buttonElement.style.background = '';

          document.removeEventListener('click', handleCancel);

        }

      };

      

      buttonElement.onclick = handleConfirm;

      setTimeout(() => document.addEventListener('click', handleCancel), 100);

    }
